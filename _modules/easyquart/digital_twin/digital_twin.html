@dataclass
class DigitalTwinEasyQuart:
    """
    A dataclass for the EASyQuart digital twin, holding the simulation data of one complete iteration.
    """

    rho_refrigerant: float | None = None  # J/(kg*K)
    """The density of the refrigerant in m³/(kg*K)"""
    cp_refrigerant: float | None = None  # m³/kg
    """The heat capacity of the refrigerant in Jkg K)"""
    timevalues: np.ndarray | None = None  # s
    """The timevalues, to which all simulation data is stored in s."""
    network_flow_rate: np.ndarray | None = None  # m³/s
    """The flow rate in m³/s of the refrigerant fluid in the main pipe of the network."""
    building_power: np.ndarray | None = None  # W
    """The needed source power of the building in W. TODO: Update later"""
    bheToNetwork_temperatures: np.ndarray | None = None  # K
    """The temperatures in K from the BHEs to the network, one value per BHE per timestep."""
    bheToNetwork_avg_temperature: np.ndarray | None = None  # K
    """The mean temperature in K of all the BHEs to the network."""
    networkToBHE_temperatures: np.ndarray | None = None  # K
    """The temperatures in K from the network to the BHEs, one value per BHE per timestep."""
    networkToBHE_avg_temperature: np.ndarray | None = None  # K
    """The mean temperature in K of the network to all BHEs."""
    networkToBuilding_temperature: np.ndarray | None = None  # K
    """The temperature in K from the network to the building."""
    buildingToNetwork_temperature: np.ndarray | None = None  # K
    """The temperature in K from the building to the network."""

    def save_h5(self, filename: Path):
        """
        Saves the class object to a hdf5-file.

        :param filename: Path of the h5-file to write to.

        :returns:
            None
        """
        with h5py.File(filename, "w") as hdf:
            # save the metadata
            hdf.attrs["class_type"] = self.__class__.__name__

            # save all attributes of the class
            for field in fields(self):
                data = getattr(self, field.name)
                if data is not None:
                    if isinstance(data, float):
                        hdf.create_dataset(field.name, data=data)
                    elif isinstance(data, int):
                        hdf.create_dataset(field.name, data=float(data))
                    else:
                        hdf.create_dataset(
                            field.name, data=data, compression="gzip"
                        )

    @classmethod
    def load_h5(cls, filename: Path) -> "DigitalTwinEasyQuart":
        """
        Load the class object from a hdf5-file.

        :param filename: Path of the h5-file to read from.

        :returns:
            a class object DigitalTwinEasyQuart
        """

        with h5py.File(filename, "r") as hdf:
            # Check the class type
            if hdf.attrs.get("class_type") != cls.__name__:
                msg = "File has not the right class type!"
                raise ValueError(msg)

            # Create a dictionary with all the data
            data = {}
            # Iterate through the attributes of the dataclass
            for field in fields(cls):
                # Check if field is present in dataset
                if field.name in hdf:
                    # Load data
                    dataset = hdf[field.name]

                    # Empty datasets as None
                    if dataset.shape is None:
                        data[field.name] = None
                    else:
                        data[field.name] = dataset[()]
                else:
                    # Not present fields as None
                    data[field.name] = None
        return cls(**data)

    def write_dataclass_to_txt(
        self, columns_to_write: list[str], path_for_txt: Path, table_name: str
    ) -> None:
        """
        Process data from the class and write it to a txt-file for Modelica input.

        :param columns_to_write: List of column names to write to the text file.
        :param path_for_txt: Path object representing the output text file.
        :param table_name: Name of the table to be called in Modelica

        :returns: None
        """
        # Create a dictionary to map column names to their units
        unit_mapping = {
            "temperature": "[K]",  # TODO: Consistency with the attribute unit or be aware of unit conversion
            "power": "[W]",
            "flow_rate": "[m³/s]",
            "time": "[s]",
        }

        # Create a DataFrame with the specified columns
        data = {}
        for column in columns_to_write:
            if "temperature" in column:
                data[column + unit_mapping["temperature"]] = getattr(
                    self, column
                )
            elif "power" in column:
                data[column + unit_mapping["power"]] = getattr(self, column)
            elif "flow_rate" in column:
                data[column + unit_mapping["flow_rate"]] = getattr(self, column)
            elif "time" in column:
                data[column + unit_mapping["time"]] = getattr(self, column)
            else:
                data[column] = getattr(self, column)

        df_data = pd.DataFrame(data)

        # Calculate the number of rows and columns
        x = len(df_data)
        y = len(df_data.columns)

        # Save DataFrame as CSV with .txt extension
        with path_for_txt.open(mode="w") as f:
            f.write("#1 \n")
            # Add first line with '#' and column names
            f.write("#" + ",".join(df_data.columns) + "\n")
            # Add second line with 'double sim_results (x, y)'
            f.write(f"double {table_name} ({x}, {y})\n")
            # Save DataFrame without index and header
            df_data.to_csv(f, index=False, header=False, lineterminator="\n")

        print(f"The file has been successfully saved as {path_for_txt}.")
