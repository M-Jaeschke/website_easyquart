    def add_variable(self, variable: UncertainVariable) -> None:
        """
        Add an UncertainVariable to the collection.

        :param variable: The variable to add.

        :raises ValueError: If a variable with the same name already exists.
        """
        if variable.name in self.variables:
            msg = f"Variable with name '{variable.name}' already exists."
            raise ValueError(msg)
        self.variables[variable.name] = variable

    def to_json(self, file_path: Path) -> None:
        """
        Write the collection of UncertainVariables to a JSON file.

        :param file_path: The path to the JSON file.
        """
        with file_path.open(mode="w") as json_file:
            json.dump(
                {
                    name: {
                        "distribution": f"{var.distribution.dist.name}({', '.join([f'{k}={v}' for k, v in var.distribution.kwds.items()])})",
                        "description": var.description,
                        "unit": var.unit,
                    }
                    for name, var in self.variables.items()
                },
                json_file,
                indent=2,
            )

    def from_json(self, file_path: Path) -> None:
        """
        Load UncertainVariables from a JSON file and add them to the collection.

        :param file_path: The path to the JSON file.
        """
        with file_path.open() as json_file:
            data = json.load(json_file)
            for name, var in data.items():
                dist_name, params = var["distribution"].split("(")
                params = params.rstrip(")")
                param_dict = {
                    k: float(v)
                    for k, v in (p.split("=") for p in params.split(", "))
                }
                dist = getattr(stats, dist_name)
                uncertain_var = UncertainVariable(
                    name=name,
                    distribution=dist(**param_dict),
                    description=var["description"],
                    unit=var["unit"],
                )
                self.add_variable(uncertain_var)

    def latin_hypercube_sampling(
        self,
        n_samples: int,
        scramble: bool = True,
        optimization: str | None = None,
        rng: np.random.Generator | None = None,
    ) -> None:
        """
        Perform Latin Hypercube Sampling and store samples in each UncertainVariable.

        :param n_samples: Number of samples to generate.
        :param scramble: If True, use scrambled sampling. Defaults to True.
        :param optimization: Optimization method. Options are None, 'random-cd' or 'lloyd'. Defaults to None.
        :param rng: Random number generator for reproducibility. Defaults to None.
        """
        d = len(self.variables)
        sampler = stats.qmc.LatinHypercube(
            d=d, scramble=scramble, optimization=optimization, rng=rng
        )
        lhs_samples = sampler.random(n=n_samples)

        for i, (_, var) in enumerate(self.variables.items()):
            var.samples = var.distribution.ppf(lhs_samples[:, i])

        """
