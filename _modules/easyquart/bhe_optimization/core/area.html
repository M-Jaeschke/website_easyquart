@dataclass
class Area:
    """
    Represents a geographic area where Borehole Heat Exchangers (BHEs) should be placed.

    Attributes:
        polygon (Polygon): The geometric representation of the area.
        d_buffer (float): Buffer distance applied inward to the area.
        obstacles (Optional[List[Obstacle]]): A list of obstacles within the area.
    """

    polygon: Polygon
    d_buffer: float = 0.0  # Default: No buffer
    obstacles: list[Obstacle] | None = None

    def __post_init__(self) -> None:
        """
        Initializes the obstacles list if not provided and ensures d_buffer is valid.
        """
        if self.obstacles is None:
            self.obstacles = []
        if self.d_buffer < 0:
            msg = "d_buffer must be non-negative."
            raise ValueError(msg)

    def buffered_polygon(self) -> Polygon:
        """
        Returns the polygon with an inward buffer applied.

        :return: A new Polygon with the buffer applied inward.
        :raises ValueError: If the buffer is too large and the polygon disappears.
        """
        shrunk_polygon = self.polygon.buffer(-self.d_buffer)

        if shrunk_polygon.is_empty:
            msg = f"Buffer {self.d_buffer} is too large, polygon disappears."
            raise ValueError(msg)

        return shrunk_polygon

    def to_json(self, filename: str) -> None:
        """
        Exports the Area instance to a JSON file using GeoJSON format.

        :param filename: Destination JSON file name.
        """
        features = [
            {
                "type": "Feature",
                "geometry": mapping(self.polygon),
                "properties": {"type": "area", "d_buffer": self.d_buffer},
            }
        ]

        # Add buffered area as a separate feature if it remains valid
        with suppress(ValueError):
            features.append(
                {
                    "type": "Feature",
                    "geometry": mapping(self.buffered_polygon()),
                    "properties": {
                        "type": "buffered_area",
                        "d_buffer": self.d_buffer,
                    },
                }
            )

        # Add obstacles
        for obstacle in self.obstacles:
            features.append(
                {
                    "type": "Feature",
                    "geometry": mapping(obstacle.polygon),
                    "properties": {"d_min": obstacle.d_min},
                }
            )

        # Save GeoJSON data
        data = {"type": "FeatureCollection", "features": features}

        path = Path(filename)
        path.parent.mkdir(parents=True, exist_ok=True)

        with path.open(mode="w", encoding="utf-8") as f:
            json.dump(data, f, indent=4, ensure_ascii=False)

    @classmethod
    def from_json(cls, filename: str) -> "Area":
        """
        Loads an Area instance from a JSON file.

        :param filename: The JSON file containing the Area data.
        :return: An Area instance.
        """
        path = Path(filename)
        if not path.exists():
            msg = f"File '{filename}' not found."
            raise FileNotFoundError(msg)

        with path.open(mode="r", encoding="utf-8") as f:
            data = json.load(f)

        if "features" not in data or len(data["features"]) == 0:
            msg = f"Invalid JSON structure in file '{filename}'."
            raise ValueError(msg)

        # Extract area polygon
        area_feature = data["features"][0]
        if (
            "geometry" not in area_feature
            or "coordinates" not in area_feature["geometry"]
        ):
            msg = "Invalid area polygon data in JSON file."
            raise ValueError(msg)

        area_polygon = Polygon(area_feature["geometry"]["coordinates"][0])
        d_buffer = area_feature["properties"].get("d_buffer", 0.0)

        # Extract obstacles
        obstacles = []
        for feature in data["features"][1:]:
            if feature["properties"].get("type") == "buffered_area":
                continue

            obstacle_polygon = Polygon(feature["geometry"]["coordinates"][0])
            d_min = feature["properties"].get("d_min", 0.25)  # Default: 0.25

            obstacles.append(Obstacle(polygon=obstacle_polygon, d_min=d_min))

        return cls(polygon=area_polygon, d_buffer=d_buffer, obstacles=obstacles)

    def create_grid(
        self, buffered_area: Polygon, resolution_meter: float
    ) -> tuple[list[tuple[int, float]], list[tuple[int, float]]]:
        """
        Generates two indexed lists:
        1) A list of (x_index, x_value) for discretized X-coordinates.
        2) A list of (y_index, y_value) for discretized Y-coordinates.

        :param buffered_area: The buffered Shapely polygon representing the area.
        :param resolution_meter: The grid resolution in meters, specifying the distance
                                between grid points along each axis.
        :return: (x_indexed, y_indexed)
                x_indexed = [(0, x0), (1, x1), ...]
                y_indexed = [(0, y0), (1, y1), ...]
        """
        minx, miny, maxx, maxy = buffered_area.bounds

        # Compute the number of grid points along the X-axis
        width_x = maxx - minx
        step_x = width_x / resolution_meter if resolution_meter > 0 else width_x
        step_x = math.ceil(step_x)  # Round up
        x_coords = np.linspace(minx, maxx, step_x + 1)
        x_indexed = list(enumerate(x_coords))

        # Compute the number of grid points along the Y-axis
        width_y = maxy - miny
        step_y = width_y / resolution_meter if resolution_meter > 0 else width_y
        step_y = math.ceil(step_y)
        y_coords = np.linspace(miny, maxy, step_y + 1)
        y_indexed = list(enumerate(y_coords))

        return x_indexed, y_indexed

    def check_valid(
        self, p: Point, buffered_area: Polygon, exclusion_zone
    ) -> bool:
        """
        Checks whether a given point is valid, meaning:
        - It is inside the buffered area.
        - It does not lie within the exclusion zone of obstacles.

        :param p: The point to be checked (Shapely Point).
        :param buffered_area: The buffered polygon defining the valid area.
        :param exclusion_zone: A unified polygon representing buffered obstacles
                            (or None if no obstacles exist).
        :return: True if the point is valid, False otherwise.
        """
        # The point must be inside the buffered area (including boundaries)
        if not buffered_area.covers(p):
            return False

        # The point must not be inside the exclusion zone
        if exclusion_zone and exclusion_zone.contains(p):
            return False

        return True

    def generate_valid_points(
        self, resolution_meter: float
    ) -> list[tuple[int, int, Point]]:
        """
        Generates a list of valid grid points within the buffered area,
        ensuring that points are not placed within buffered obstacles.

        :param resolution_meter: Grid resolution in meters, determining the
                                spacing between grid points.
        :return: A list of (x_index, y_index, Point(x, y)) tuples representing
                valid grid points.
        """
        buffered_area = self.buffered_polygon()

        # Create exclusion zone by buffering all obstacles
        buffered_obstacles = [
            obs.polygon.buffer(obs.d_min) for obs in self.obstacles
        ]
        exclusion_zone = (
            unary_union(buffered_obstacles) if buffered_obstacles else None
        )

        # 1) Generate discretized coordinates using create_grid()
        x_indexed, y_indexed = self.create_grid(buffered_area, resolution_meter)

        # 2) Iterate through all (index, coordinate) pairs for X and Y
        valid_points = []
        for i, x_val in x_indexed:
            for j, y_val in y_indexed:
                p = Point(x_val, y_val)
                # Check validity
                if self.check_valid(p, buffered_area, exclusion_zone):
                    valid_points.append((i, j, p))

        return valid_points



    def __repr__(self) -> str:
        """
        Returns a 'readable' version of the Area-Instance, for debugging purposes.
        """

        return f"Area(vertices={list(self.polygon.exterior.coords)}, obstacles={len(self.obstacles)})"
